% ============================================================================
% ENGINEERING REPORT — BRANDING PREAMBLE (Option A: include-in-header)
% This file is NOT parsed as a Quarto template, so raw LaTeX (braces, \@...)
% is safe here. Variable values (\metaTitle, \metaAuthor, etc.) are provided
% by before-body.tex via \renewcommand — see that file.
% Quarto already loads: longtable, booktabs, array, calc, graphicx, hyperref,
% the syntax-highlighting macros, \tightlist, \passthrough, \pandocbounded.
% We therefore only declare packages specific to OUR layout.
% ============================================================================

\usepackage[table]{xcolor}
\arrayrulecolor{white}

% ============================================================================
% FONT — TRUE ARIAL FOR EVERYTHING (requires pdf-engine: xelatex)
% fontspec loads the real system font. Setting BOTH the main and sans families
% to Arial means body text, headings, and any \sffamily (your headers/footers)
% all render in Arial. \familydefault is left as the default; fontspec handles
% the mapping, so no \renewcommand override is needed.
% ============================================================================
\usepackage{fontspec}
\setmainfont{Arial}
\setsansfont{Arial}
\setmonofont{Courier New}   % monospace for code/Python output (any mono is fine)

% ----------------------------------------------------------------------------
% Resolve the logo by filename from the templates/ folder (and project root).
% ----------------------------------------------------------------------------
\graphicspath{{templates/}{./}}

% ----------------------------------------------------------------------------
% PAGE GEOMETRY (unchanged from your design)
% ----------------------------------------------------------------------------
\usepackage[
top=60mm,
bottom=30mm,
left=25mm,
right=25mm,
headheight=110pt,
headsep=50pt
]{geometry}

\setlength{\topskip}{15pt}

\usepackage{fancyhdr}
\usepackage{tabularx}

% ============================================================================
% TITLE BLOCK — printed in the body of the title page (standard layout:
% title, subtitle, author, date), beneath the branded header band.
% ============================================================================
